Add a component definition based on the Reference CPU*

  1. In your working folder, open components.xml with a text editor.

    Tip

    Observe the predefined components.

  2. Toward the end of the file, before the line, </group>, insert the component definition: 

     <item name="Ref Des CPU*">
    
      <!-- Match components with package starting with ces
           and DIP14 package -->
      <record name="Reference">
        <text value="CPU*"/>
      </record>
      
      <!-- Resistor model:  "0", "1", or "2" -->
      <record name="Resistor Model">
        <int value="2"/>
      </record>
      
      <!-- Model parameters -->
      <record name="Theta JC">
        <float value="0.4"/>
      </record>
      <record name="Theta JB">
        <float value="0.8"/>
      </record>
      <record name="Load">
        <float value="2.5"/>
      </record>
      <record name="Maximum Junction Temperature">
        <float value="120"/>
      </record>
      <record name="Maximum Case Temperature">
        <float value="100"/>
      </record>
      
      <!-- Emissivity option:
             "0" = Use Thermo-Optical Properties
             "1" = Specify -->
      <record name="Emissivity Setting">
        <int value="0"/>
      </record>
      
      <!-- Additional parameters
             Use a bool value to control a toggle
             Use an int value to control a menu -->
      <record name="Single Footprint">
        <bool value="0"/>
      </record>
      <record name="Coupling Resolution">
        <int value="1"/>
      </record>
      <record name="Report">
        <bool value="1"/>
      </record>
      <record name="Track During Solve">
        <bool value="0"/>
      </record>
      
      <!-- Settings specific to PCB Exchange
           Controls how the component is meshed automatically
           If not specified, default settings are used -->
      <record name="Element Size">
        <float value="0.0075"/>
      </record>
	  <record name="Element Thickness">
	    <float value="0.005"/>
	  </record>
      <record name="Element Color">
        <text value="yellow"/>
      </record>
      <record name="Case Material">
        <text value="Plastic"/>
      </record>
    </item>

  3. Save and close the components.xml file.
